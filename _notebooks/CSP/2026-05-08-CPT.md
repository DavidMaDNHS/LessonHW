<script type="module">

import { queryGemini } from '{{ site.baseurl }}/assets/js/api/gemini.js';

const ANALYSIS_PROMPTS = {
    "cpt-check":
        "Please analyze this APCSP CPT project and identify whether it includes the required CPT components such as input, output, lists, procedures, and algorithms:",

    "idea-check":
        "Please analyze this APCSP CPT idea and provide feedback on simplicity, feasibility, and CPT compatibility:",

    "requirement-check":
        "Please verify whether this APCSP CPT explanation correctly addresses the required CPT components:",

    "frq-help":
        "Please help answer this APCSP CPT FRQ and explain the concepts clearly:"
};

document.addEventListener("DOMContentLoaded", function() {

    const root = document.getElementById("cpt-guide-page");

    var quill = new Quill('#cpt-quill-editor', {
        theme: 'snow'
    });

    loadRandomSample();

    document.getElementById("cpt-test-mode").onclick = function() {

        const sampleText = `
My CPT project is a school club finder.

The user types a club name into the input box.

The program searches through a list of clubs using a loop and if statement.

If a match is found, the meeting information is displayed to the user.

The program uses:
- Input
- Output
- List
- Procedure with parameter
- Algorithm with sequencing, selection, and iteration
`;

        quill.setText(sampleText);

        showStatusMessage(
            "🧪 Test mode loaded successfully!",
            "info"
        );
    };

    document.getElementById("cpt-saveBtn").onclick = function() {

        const draft = {
            content: quill.getContents(),
            plainText: quill.getText(),
            mode: document.getElementById("cpt-analysisMode").value,
            timestamp: new Date().toISOString()
        };

        try {

            localStorage.setItem(
                'cpt-writing-draft',
                JSON.stringify(draft)
            );

            showStatusMessage(
                "✅ Draft saved successfully!",
                "success"
            );

        } catch (error) {

            showStatusMessage(
                "❌ Failed to save draft",
                "error"
            );
        }
    };

    document.getElementById("cpt-loadBtn").onclick = function() {

        try {

            const savedDraft = localStorage.getItem('cpt-writing-draft');

            if(savedDraft){

                const draft = JSON.parse(savedDraft);

                quill.setContents(draft.content);

                document.getElementById("cpt-analysisMode").value = draft.mode;

                showStatusMessage(
                    "✅ Draft loaded successfully!",
                    "success"
                );

            } else {

                showStatusMessage(
                    "⚠️ No saved draft found",
                    "warning"
                );
            }

        } catch(error){

            showStatusMessage(
                "❌ Failed to load draft",
                "error"
            );
        }
    };

    function showStatusMessage(message, type){

        const statusDiv = document.getElementById("cpt-status-message");

        statusDiv.textContent = message;

        statusDiv.style.display = "block";

        switch(type){

            case "success":
                statusDiv.style.backgroundColor = "#d4edda";
                statusDiv.style.color = "#155724";
                break;

            case "error":
                statusDiv.style.backgroundColor = "#f8d7da";
                statusDiv.style.color = "#721c24";
                break;

            case "warning":
                statusDiv.style.backgroundColor = "#fff3cd";
                statusDiv.style.color = "#856404";
                break;

            case "info":
                statusDiv.style.backgroundColor = "#d1ecf1";
                statusDiv.style.color = "#0c5460";
                break;
        }

        setTimeout(() => {
            statusDiv.style.display = "none";
        }, 3000);
    }

    document.getElementById("cpt-checkBtn").onclick = function(){

        const text = quill.getText();

        const mode = document.getElementById("cpt-analysisMode").value;

        const outputDiv = document.getElementById("cpt-output");

        outputDiv.textContent = "⏳ Analyzing...";

        const prompt =
            ANALYSIS_PROMPTS[mode] ||
            ANALYSIS_PROMPTS["cpt-check"];

        queryGemini({
            prompt: prompt,
            text: text
        })

        .then(result => {

            const markdown = result.text;

            const htmlContent = marked.parse(markdown);

            outputDiv.innerHTML = htmlContent;

            showStatusMessage(
                "✅ Analysis completed!",
                "success"
            );
        })

        .catch(error => {

            outputDiv.textContent = "";

            showStatusMessage(
                "⚠️ " + error.message,
                "error"
            );
        });
    };

    function loadRandomSample(){

        const mode =
            document.getElementById("cpt-analysisMode").value;

        const samples =
            root.querySelectorAll(
                `.sample-text[data-type="${mode}"]`
            );

        if(samples.length === 0) return;

        const randomIndex =
            Math.floor(Math.random() * samples.length);

        quill.setText(
            samples[randomIndex].textContent.trim()
        );
    }

    document.getElementById("cpt-analysisMode").onchange =
        function(){
            loadRandomSample();
        };

});

</script>
